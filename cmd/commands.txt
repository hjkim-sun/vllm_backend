# uv & 라이브러리 설치
curl -LsSf https://astral.sh/uv/install.sh | sh
cd /workspace
uv init
uv add vllm==0.11.0 transformers==4.57.6 torch==2.8.0 hf_transfer 
mkdir model_repository

# 모델 로드 
uv run vllm serve openai/gpt-oss-20b \
--download-dir /workspace/model_repository \
--api-key myvllmserver

# 로드 상태 확인
curl http://localhost:8000/v1/models -H "Authorization: Bearer myvllmserver" 

# 메시지 요청 (맥북)
curl -X POST https://{runpod 주소}/v1/chat/completions -H "Content-Type: application/json" -H "Authorization: Bearer myvllmserver" -d '{"messages": [{"role": "user", "content":"안녕, 너는 누구야?"}], "temperature": 0.7}'
